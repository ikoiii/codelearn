Pengenalan JavaScript

JavaScript adalah bahasa pemrograman yang digunakan untuk membuat halaman web interaktif dan dinamis.

Apa itu JavaScript?

  • JavaScript adalah bahasa pemrograman client-side
  • JavaScript membuat website interaktif dan responsif
  • JavaScript dapat memanipulasi HTML dan CSS
  • JavaScript dapat menghandle event user (klik, hover, dll)
  • JavaScript dapat berkomunikasi dengan server (AJAX/Fetch)

Cara Menggunakan JavaScript:

1. Inline JavaScript

<button onclick="alert('Hello World!')">Klik Saya</button>

2. Internal JavaScript

<script> function sayHello() { alert('Hello World!'); } </script>

3. External JavaScript (Recommended)

<!-- Di HTML --> <script src="script.js"></script> <!-- Di file script.js --> function sayHello() { alert('Hello World!'); }

Dasar-dasar JavaScript:

Variables (Variabel)

// ES6 - Modern JavaScript let name = "John Doe"; // Bisa diubah const age = 25; // Tidak bisa diubah var isStudent = true; // Cara lama (hindari) // Variable naming let userName = "alice"; // camelCase let user_name = "bob"; // snake_case let USER_NAME = "charlie"; // UPPER_CASE (konstanta)

Data Types

// Primitive types let text = "Hello World"; // String let number = 42; // Number let decimal = 3.14; // Number let isTrue = true; // Boolean let isFalse = false; // Boolean let nothing = null; // Null let notDefined; // Undefined // Reference types let person = { // Object name: "John", age: 25 }; let colors = ["red", "green", "blue"]; // Array

Operators

// Arithmetic operators let x = 10; let y = 3; console.log(x + y); // 13 (penjumlahan) console.log(x - y); // 7 (pengurangan) console.log(x * y); // 30 (perkalian) console.log(x / y); // 3.33 (pembagian) console.log(x % y); // 1 (modulus/sisa bagi) // Comparison operators console.log(x == y); // false (sama nilai) console.log(x === y); // false (sama nilai dan tipe) console.log(x != y); // true (tidak sama nilai) console.log(x !== y); // true (tidak sama nilai atau tipe) console.log(x > y); // true (lebih besar) console.log(x < y); // false (lebih kecil) // Logical operators let a = true; let b = false; console.log(a && b); // false (AND) console.log(a || b); // true (OR) console.log(!a); // false (NOT)

Control Structures

If-Else Statement

let age = 18; if (age >= 18) { console.log("Anda dewasa"); } else if (age >= 13) { console.log("Anda remaja"); } else { console.log("Anda anak-anak"); } // Ternary operator let message = age >= 18 ? "Dewasa" : "Anak-anak";

Switch Statement

let day = "Monday"; switch (day) { case "Monday": console.log("Hari Senin"); break; case "Tuesday": console.log("Hari Selasa"); break; default: console.log("Hari lainnya"); }

Loops (Perulangan)

// For loop for (let i = 0; i < 5; i++) { console.log("Iterasi ke-" + i); } // While loop let count = 0; while (count < 5) { console.log("Count: " + count); count++; } // For...of loop (untuk array) let colors = ["red", "green", "blue"]; for (let color of colors) { console.log(color); } // For...in loop (untuk object) let person = { name: "John", age: 25 }; for (let key in person) { console.log(key + ": " + person[key]); }

Functions

// Function declaration function greet(name) { return "Hello, " + name + "!"; } // Function expression const sayHello = function(name) { return "Hi, " + name; }; // Arrow function (ES6) const welcome = (name) => { return "Welcome, " + name; }; // Arrow function singkat const add = (a, b) => a + b; // Menggunakan function console.log(greet("Alice")); console.log(sayHello("Bob")); console.log(welcome("Charlie")); console.log(add(5, 3));

Arrays

// Membuat array let fruits = ["apple", "banana", "orange"]; let numbers = [1, 2, 3, 4, 5]; let mixed = ["text", 42, true, null]; // Array methods fruits.push("grape"); // Tambah di akhir fruits.unshift("mango"); // Tambah di awal fruits.pop(); // Hapus di akhir fruits.shift(); // Hapus di awal // Mengakses elemen console.log(fruits[0]); // "apple" console.log(fruits.length); // Panjang array // Array methods modern let doubled = numbers.map(n => n * 2); // [2, 4, 6, 8, 10] let evens = numbers.filter(n => n % 2 === 0); // [2, 4] let sum = numbers.reduce((a, b) => a + b, 0); // 15

Objects

// Membuat object let person = { firstName: "John", lastName: "Doe", age: 25, isStudent: true, hobbies: ["reading", "coding", "gaming"], // Method dalam object greet: function() { return "Hello, I'm " + this.firstName; } }; // Mengakses properties console.log(person.firstName); // "John" console.log(person["lastName"]); // "Doe" console.log(person.greet()); // "Hello, I'm John" // Menambah property person.email = "john@example.com"; // Menghapus property delete person.isStudent;

DOM Manipulation:

// Mengambil element const element = document.getElementById('myId'); const elements = document.getElementsByClassName('myClass'); const elements = document.querySelectorAll('.mySelector'); // Mengubah konten element.textContent = "New text content"; element.innerHTML = "Bold text"; // Mengubah style element.style.color = "red"; element.style.fontSize = "18px"; // Menambah event listener element.addEventListener('click', function() { alert('Element clicked!'); });

Latihan:

Coba buat fungsi JavaScript sederhana:

// Kalkulator sederhana function calculator(operator, a, b) { switch (operator) { case '+': return a + b; case '-': return a - b; case '*': return a * b; case '/': return a / b; default: return "Operator tidak valid"; } } // Penggunaan console.log(calculator('+', 5, 3)); // 8 console.log(calculator('*', 4, 7)); // 28
